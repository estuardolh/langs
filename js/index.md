Operators
```javascript
// math
a % b // modulus
a *= b // a = a * b

// comparation
=== // equal value and equal type
== // equal
!= // not equals
! // NOT
```

Expressions
```javascript
typeof value; // return type.

( a == b ? value_if_true : value_if_else ); // conditional

// optional args
function optional_param( param ){
  return typeof param === 'undefined';
}
optional_param(2); // return false
optional_param(); // return true
```

Math
```javascript
// parse to int
parseInt( number );

// distance between ...
Math.sqrt( Math.pow(x1 - x2 , 2) + Math.pow( y1 - y2, 2 ) );

// random number between 0 and 1(?)
Math.random();

// radians to degrees
var degrees = radians * ( 180 / Math.PI );

// function: angle to [x, y]
return [ Math.cos( radians ) * magnitude, Math.sin( radians ) *  magnitude ];

// function: (x, y) to angle
var radians = Math.atan( x / y );
var degrees = radians * ( 180 / Math.PI );
return ( degrees < 0 ? 360 - degrees : degrees );
```
